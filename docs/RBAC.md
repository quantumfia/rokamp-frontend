### 데이터 필터링 원칙

```typescript
// 역할별 부대 필터링 로직 예시
function getAccessibleUnits(userRole: UserRole, userUnit: string): string[] {
  switch (userRole) {
    case 'ROLE_HQ':
      return getAllUnits(); // 전군
    case 'ROLE_DIV':
      return getSubordinateUnits(userUnit); // 예하 부대
    case 'ROLE_BN':
      return [userUnit]; // 본인 부대만
  }
}
```
